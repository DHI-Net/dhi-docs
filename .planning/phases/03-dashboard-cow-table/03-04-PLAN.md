```javascript
import { computed } from 'vue'
import { storeToRefs } from 'pinia'
import { useFarmStore } from '../stores/farm.js'
import { useFarmData } from '../composables/useFarmData.js'
import { ALERT_CONFIG } from '../utils/alerts.js'

const farmStore = useFarmStore()
const { currentFarm, currentMonth } = storeToRefs(farmStore)
const { cows, loading } = useFarmData()

const ALERT_METRICS = {
  '(A)': { metricKey: 'fat', metricLabel: '乳脂肪', metricHighlight: 'text-yellow-700' },
  '(B)': { metricKey: 'scc', metricLabel: '體細胞數', metricHighlight: 'text-red-700' },
  '(C)': { metricKey: 'milk', metricLabel: '乳量', metricHighlight: 'text-orange-700' },
  '(D)': { metricKey: 'dim', metricLabel: '泌乳天數', metricHighlight: 'text-blue-700' },
  '(E)': { metricKey: 'ffa', metricLabel: '游離脂肪酸', metricHighlight: 'text-purple-700' },
}

const ALERT_ORDER = ['(B)', '(C)', '(E)', '(A)', '(D)']  // ordered by severity/importance

const alertGroups = computed(() => {
  if (!cows.value || cows.value.length === 0) return []

  const groups = {}
  for (const cow of cows.value) {
    if (!cow.alerts || !Array.isArray(cow.alerts)) continue
    for (const code of cow.alerts) {
      if (!ALERT_CONFIG[code]) continue
      if (!groups[code]) groups[code] = []
      groups[code].push(cow)
    }
  }

  return ALERT_ORDER
    .filter(code => groups[code]?.length > 0)
    .map(code => ({
      code,
      label: ALERT_CONFIG[code].label,
      badgeClass: ALERT_CONFIG[code].color,
      cows: groups[code],
      ...ALERT_METRICS[code],
    }))
})

function fmt(v) {
  if (v == null) return '--'
  return typeof v === 'number' ? v.toFixed(1) : String(v)
}
```

Key points:
- Groups are ordered by severity: (B) SCC first (most critical), then (C), (E), (A), (D)
- Each group shows alert badge with count, then a mini-table of cows
- The "relevant metric" column changes per alert type (fat for A, scc for B, etc.)
- Uses existing ALERT_CONFIG from alerts.js for consistent color coding
- No pagination needed per Claude's Discretion in CONTEXT.md (cows per alert are usually few)
