```javascript
import { defineStore } from 'pinia'
import { ref, reactive } from 'vue'

const STORAGE_KEY = 'dhi_overview_presets'

function loadPresetsFromStorage() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY)
    return raw ? JSON.parse(raw) : []
  } catch {
    return []
  }
}

function savePresetsToStorage(presets) {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(presets))
}

export const useOverviewFilterStore = defineStore('overviewFilter', () => {
  // --- Filter state (per D-05) ---
  const selectedFarms = ref(new Set())      // empty = all farms
  const timeRange = ref(12)                  // 3 | 6 | 12 | 'custom'
  const startDate = ref(null)                // 'YYYY.M' | null
  const endDate = ref(null)                  // 'YYYY.M' | null
  const thresholds = reactive({ scc: null, milk: null, open_days: null })
  const alertTypes = ref(new Set())          // Set<string> e.g. '(A)', '(B)'

  // --- Saved presets (per D-06) ---
  const savedPresets = ref(loadPresetsFromStorage())

  // --- Actions ---
  function setSelectedFarms(farms) {
    selectedFarms.value = new Set(farms)
  }

  function toggleFarm(farmName) {
    const s = new Set(selectedFarms.value)
    if (s.has(farmName)) { s.delete(farmName) } else { s.add(farmName) }
    selectedFarms.value = s
  }

  function setTimeRange(value) {
    timeRange.value = value
  }

  function setCustomRange(start, end) {
    startDate.value = start
    endDate.value = end
    timeRange.value = 'custom'
  }

  function setThreshold(metric, value) {
    thresholds[metric] = value
  }

  function clearThresholds() {
    thresholds.scc = null
    thresholds.milk = null
    thresholds.open_days = null
  }

  function setAlertTypes(types) {
    alertTypes.value = new Set(types)
  }

  function resetAll() {
    selectedFarms.value = new Set()
    timeRange.value = 12
    startDate.value = null
    endDate.value = null
    clearThresholds()
    alertTypes.value = new Set()
    // Note: savedPresets NOT cleared (per D-06)
  }

  // --- Preset management (per D-06) ---
  function _snapshotFilters() {
    return {
      selectedFarms: [...selectedFarms.value],
      timeRange: timeRange.value,
      startDate: startDate.value,
      endDate: endDate.value,
      thresholds: { ...thresholds },
      alertTypes: [...alertTypes.value],
    }
  }

  function savePreset(name) {
    const snapshot = _snapshotFilters()
    const savedAt = new Date().toISOString()
    const existing = savedPresets.value.findIndex(p => p.name === name)
    if (existing >= 0) {
      savedPresets.value[existing] = { name, filters: snapshot, savedAt }
    } else {
      savedPresets.value.push({ name, filters: snapshot, savedAt })
    }
    // Trigger reactivity
    savedPresets.value = [...savedPresets.value]
    savePresetsToStorage(savedPresets.value)
  }

  function applyPreset(name) {
    const preset = savedPresets.value.find(p => p.name === name)
    if (!preset) return
    const f = preset.filters
    selectedFarms.value = new Set(f.selectedFarms || [])
    timeRange.value = f.timeRange ?? 12
    startDate.value = f.startDate ?? null
    endDate.value = f.endDate ?? null
    thresholds.scc = f.thresholds?.scc ?? null
    thresholds.milk = f.thresholds?.milk ?? null
    thresholds.open_days = f.thresholds?.open_days ?? null
    alertTypes.value = new Set(f.alertTypes || [])
  }

  function deletePreset(name) {
    savedPresets.value = savedPresets.value.filter(p => p.name !== name)
    savePresetsToStorage(savedPresets.value)
  }

  return {
    selectedFarms, timeRange, startDate, endDate, thresholds, alertTypes, savedPresets,
    setSelectedFarms, toggleFarm, setTimeRange, setCustomRange,
    setThreshold, clearThresholds, setAlertTypes,
    resetAll, savePreset, applyPreset, deletePreset,
  }
})
```
