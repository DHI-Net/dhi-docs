```javascript
import '../utils/echarts.js'
import { ref, computed } from 'vue'
import { storeToRefs } from 'pinia'
import VChart from 'vue-echarts'
import { useOverviewFilterStore } from '../stores/overviewFilter.js'
import { sortMonthKeys, sliceTrendByRange, assembleTrendData } from '../composables/useTrendData.js'

const overviewFilter = useOverviewFilterStore()
const { selectedFarms, timeRange, startDate, endDate } = storeToRefs(overviewFilter)

const COLORS = ['#5470c6', '#91cc75', '#fac858', '#ee6666', '#73c0de']

const timeOptions = [
  { value: 3, label: '近 3 個月' },
  { value: 6, label: '近 6 個月' },
  { value: 12, label: '近 12 個月' },
]

// Extract all farm data from window.__DHI_DATA__
const staticData = window.__DHI_DATA__ || {}
const allFarmNames = Object.keys(staticData)

// Build per-farm trend data, filtered by selectedFarms and timeRange
const farmTrends = computed(() => {
  const farms = selectedFarms.value.size === 0
    ? allFarmNames
    : allFarmNames.filter(f => selectedFarms.value.has(f))

  return farms.map(farmName => {
    const farmData = staticData[farmName]
    if (!farmData?.months) return { name: farmName, trend: { months: [], milk: [], scc: [], count: [] } }

    const monthSummaries = {}
    for (const [monthKey, monthData] of Object.entries(farmData.months)) {
      if (monthData?.summary) {
        monthSummaries[monthKey] = monthData.summary
      }
    }

    const fullTrend = assembleTrendData(monthSummaries)
    const sliced = sliceTrendByRange(fullTrend, timeRange.value, startDate.value, endDate.value)
    return { name: farmName, trend: sliced }
  })
})

// Unified X-axis: all unique months across all farms, sorted
const unifiedMonths = computed(() => {
  const allMonths = new Set()
  for (const ft of farmTrends.value) {
    for (const m of ft.trend.months) allMonths.add(m)
  }
  return sortMonthKeys([...allMonths])
})

// Build chart option for a given metric key ('milk', 'scc', 'count')
function buildChartOption(metricKey, yAxisName) {
  return computed(() => {
    const months = unifiedMonths.value
    const series = farmTrends.value.map((ft, idx) => {
      // Align farm data to unified X-axis
      const dataMap = {}
      for (let i = 0; i < ft.trend.months.length; i++) {
        dataMap[ft.trend.months[i]] = ft.trend[metricKey][i]
      }
      return {
        name: ft.name,
        type: 'line',
        data: months.map(m => dataMap[m] ?? null),
        itemStyle: { color: COLORS[idx % COLORS.length] },
        connectNulls: true,
      }
    })
    return {
      tooltip: { trigger: 'axis' },
      legend: { data: farmTrends.value.map(f => f.name), top: 0 },
      grid: { left: 60, right: 20, bottom: 30, top: 40 },
      xAxis: { type: 'category', data: months },
      yAxis: { type: 'value', name: yAxisName },
      series,
    }
  })
}

const milkChartOption = buildChartOption('milk', '乳量 (kg)')
const countChartOption = buildChartOption('count', '頭數')
const sccChartOption = buildChartOption('scc', 'SCC (萬/mL)')
```

1. **Create `frontend/src/composables/useComparisonData.js`**:

This composable extracts current vs previous month comparison data from the summary object.

```javascript
import { computed } from 'vue'

const METRICS = [
  { key: 'milk', label: '平均乳量', unit: 'kg' },
  { key: 'scc', label: '平均體細胞數', unit: '萬/mL' },
  { key: 'count', label: '泌乳頭數', unit: '頭' },
  { key: 'open_days', label: '平均空胎天數', unit: '天' },
  { key: 'urea', label: '平均尿素氮', unit: 'mg/dL' },
  { key: 'pf', label: '平均 P/F', unit: '' },
]

export function useComparisonData(summary) {
  const comparisonMetrics = computed(() => {
    if (!summary.value?.comparison) return []
    const cmp = summary.value.comparison
    return METRICS.map(m => {
      const curr = cmp[m.key]?.curr ?? null
      const prev = cmp[m.key]?.prev ?? null
      return { ...m, curr, prev }
    }).filter(m => m.curr !== null || m.prev !== null)
  })

  return { comparisonMetrics, METRICS }
}

import '../utils/echarts.js'
import { computed } from 'vue'
import { storeToRefs } from 'pinia'
import VChart from 'vue-echarts'
import { useFarmStore } from '../stores/farm.js'
import { useFarmData } from '../composables/useFarmData.js'
import { useComparisonData } from '../composables/useComparisonData.js'

const farmStore = useFarmStore()
const { currentFarm, currentMonth } = storeToRefs(farmStore)
const { summary, loading } = useFarmData()
const { comparisonMetrics } = useComparisonData(summary)

// Grouped bar chart: 本月 (brand-600) vs 上月 (gray-300)
const barChartOption = computed(() => {
  const labels = comparisonMetrics.value.map(m => m.label)
  const prevData = comparisonMetrics.value.map(m => m.prev)
  const currData = comparisonMetrics.value.map(m => m.curr)

  return {
    tooltip: { trigger: 'axis' },
    legend: { data: ['上月', '本月'], top: 0 },
    grid: { left: 80, right: 20, bottom: 30, top: 40 },
    xAxis: { type: 'category', data: labels },
    yAxis: { type: 'value' },
    series: [
      { name: '上月', type: 'bar', data: prevData, itemStyle: { color: '#d1d5db' } },
      { name: '本月', type: 'bar', data: currData, itemStyle: { color: '#16a34a' } },
    ],
  }
})

function fmt(v) {
  return v != null ? Number(v).toFixed(1) : '--'
}

function fmtDelta(m) {
  if (m.curr == null || m.prev == null) return '--'
  const d = m.curr - m.prev
  return `${d > 0 ? '+' : ''}${d.toFixed(1)}`
}

function deltaClass(m) {
  if (m.curr == null || m.prev == null) return 'text-gray-400'
  const d = m.curr - m.prev
  if (d === 0) return 'text-gray-400'
  // For SCC and open_days, lower is better
  const negativePolarity = ['scc', 'open_days']
  const isImproved = negativePolarity.includes(m.key) ? d < 0 : d > 0
  return isImproved ? 'text-green-600' : 'text-red-500'
}
```
